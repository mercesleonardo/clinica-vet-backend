symfony new my_project_directory --version="7.3.x"

################################ Gemini
# O "cérebro" para criar APIs rapidamente (CRUD, filtros, etc.)
composer require api-platform/api-platform

# Para gerar código (entidades, controllers, etc.)
composer require symfony/maker-bundle

# Para autenticação e hash de senha (bcrypt já é o padrão)
composer require symfony/security-bundle

# Para gerar tokens JWT no login
composer require lexik/jwt-authentication-bundle

# Para permitir que o frontend (ex: localhost:5173) acesse a API (ex: localhost:8000)
composer require nelmio/cors-bundle

php bin/console lexik:jwt:generate-keys
